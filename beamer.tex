    \begin{theorem}[Teorema del Sándwich]
    Sea $f$, $g$ y $h$ funciones tales que $g(x) \leq f(x) \leq h(x)$ para todo $x \in [a,b]$, y supongamos que
    \[
    \lim_{x \to c} g(x) = \lim_{x \to c} h(x) = L
    \]
    para algún $c \in (a,b)$. Entonces, 
    \[
    \lim_{x \to c} f(x) = L.
    \]
    \end{theorem}
